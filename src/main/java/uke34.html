Enkel kalkulator

Skriv et Python-program som tar inn to tall fra brukeren og utfører en av de fire grunnleggende regneoperasjonene (addisjon, subtraksjon, multiplikasjon eller divisjon) på tallene. Programmet skal deretter skrive ut resultatet.

                
# -*- coding: utf-8 -*-

def main():
    num1 = float(input("Skriv inn det første tallet: "))
    operator = input("Skriv inn operatøren (+, -, *, /): ")
    num2 = float(input("Skriv inn det andre tallet: "))

    if operator == "+":
        result = num1 + num2
    elif operator == "-":
        result = num1 - num2
    elif operator == "*":
        result = num1 * num2
    elif operator == "/":
        result = num1 / num2
    else:
        print("Ugyldig operatør!")
        return

    print("Resultatet er:", result)

if __name__ == "__main__":
    main()

Samme oppgave på nytt, men denne gangen med språket Java.

                
        import java.util.Scanner;

        public class kalkulator {
        public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Skriv inn det første tallet: ");
        double num1 = scanner.nextDouble();

        System.out.print("Skriv inn operatøren (+, -, *, /): ");
        char operator = scanner.next().charAt(0);

        System.out.print("Skriv inn det andre tallet: ");
        double num2 = scanner.nextDouble();

        double result = 0;

        switch (operator) {
            case '+':
                result = num1 + num2;
                break;
            case '-':
                result = num1 - num2;
                break;
            case '*':
                result = num1 * num2;
                break;
            case '/':
                result = num1 / num2;
                break;
            default:
                System.out.println("Ugyldig operatør!");
        }

        System.out.println("Resultatet er: " + result);
    }
}
